#상속:공통된 클레스 있고 조금다른 클레스을 만들고싶을떄

class person:
    def __init__(self,name,age):
        self.name=name
        self.age=age#initialize(초기화)의약자 self에서 첫인자받고 neme에서 새로쓸인자 받아
    def say_Hello(self, to_name):
        print("안녕 나는"+to_name+ " 나는"+ self.name)
    def introduce(self):
        print("내 이름은 "+ self.name+" 그리고 나는"+str(self.age)+"살이야")

class police(person):  #person에서 정의된 함수들이 police에 그대로 상속!!
    def arrest(self,to_arrest):
        print("넌 체포됐다"+to_arrest)
class programer(person):
    def program(selfself,to_program):
        print("다음엔 뭘 만들지? 아 이걸 만들어야겠다"+to_program)
wonie= person("워니",20)
jenny= police("제니",21)
mike= programer("마이클",22)
wonie.introduce()
jenny.arrest("ㅉㅉ")
mike.program("이메일클라이언트")

class police(person):
    def arrest(self,to_arrest):
        print("넌 체포됐다"+to_arrest)
